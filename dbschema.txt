Users
    user_id INT autoincrement
    email VARCHAR(200)
    first_name VARCHAR(20)
    last_name VARCHAR(20)
    username VARCHAR(20)
    password VARCHAR(20)

Memberships
    memberShip_id INT
    project_id INT , foreign key to Project
    member_id INT , foreign key to Member

Project
    project_id INT
    project_name VARCHAR(100), name of project
    project_description TEXT

Task
    task_id INT
    project_id INT
    user_id INT
    due_date DATE
    description TEXT
